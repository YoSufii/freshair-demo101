<script>
/* === WEBHOOK: inserted as requested === */
const webhook = "https://hooks.zapier.com/hooks/catch/26818877/uxn7mrh/";

/* helper */
function escapeHtml(s){ return String(s||'').replace(/[&<>"']/g,c=>({ '&':'&amp;','<':'&lt;','>':'&gt;','"':'&quot;',"'":'&#39;'}[c])); }

/* UI elements */
const form = document.getElementById('leadForm');
const status = document.getElementById('status');
const leadsContainer = document.getElementById('leadsContainer');
const submitBtn = document.getElementById('submitBtn');

/* load session leads */
function loadSessionLeads(){
  leadsContainer.innerHTML='';
  const arr = JSON.parse(localStorage.getItem('freshair_demo_leads') || '[]').slice().reverse();
  if(!arr.length){ leadsContainer.innerHTML = '<div class="small">No leads yet — submit one to test.</div>'; return; }
  arr.forEach(l=>{
    const el = document.createElement('div'); el.className='lead';
    el.innerHTML = `<strong>${escapeHtml(l.name)}</strong> <div class="small">${escapeHtml(l.email)} • ${escapeHtml(l.location)}</div>`;
    leadsContainer.appendChild(el);
  })
}
loadSessionLeads();

form.addEventListener('submit', async (e)=>{
  e.preventDefault();
  status.textContent = 'Sending…';
  submitBtn.disabled = true;

  const data = new FormData(form);
  const payload = {};
  data.forEach((v,k)=> payload[k]=v);

  try{
    const res = await fetch(webhook, {
      method: 'POST',
      headers: {'Content-Type':'application/json'},
      body: JSON.stringify(payload)
    });

    // save locally for demo display
    const arr = JSON.parse(localStorage.getItem('freshair_demo_leads') || '[]');
    arr.push(payload);
    localStorage.setItem('freshair_demo_leads', JSON.stringify(arr));
    loadSessionLeads();

    if (!res.ok) {
      status.textContent = 'Sent (warning: webhook returned ' + res.status + ')';
    } else {
      status.textContent = 'Sent — showing on site';
    }

    // clear & redirect to thank you quickly
    form.reset();
    setTimeout(()=> window.location.href = 'thankyou.html', 700);

  } catch(err){
    console.error(err);
    status.textContent = 'Error sending — check console';
  } finally {
    submitBtn.disabled = false;
    setTimeout(()=> status.textContent = '', 3000);
  }
});
</script>
