/* load recent leads from localStorage */
function loadLeadsUI(){
  const node = document.getElementById('recentLeads');
  node.innerHTML='';
  const leads = JSON.parse(localStorage.getItem('freshair_demo_leads')||'[]').slice().reverse();
  if(!leads.length){ node.innerHTML = '<div class="small">No demo leads yet — submit one from "Request Consultation".</div>'; return; }
  leads.forEach(l=>{
    const d = document.createElement('div');
    d.className='lead';
    d.innerHTML = `<strong>${escapeHtml(l.name||'—')}</strong> <div class="small">${escapeHtml(l.email||'—')} • ${escapeHtml(l.location||'—')}</div>`;
    node.appendChild(d);
  })
}
function escapeHtml(s){ return String(s||'').replace(/[&<>"']/g,c=>({ '&':'&amp;','<':'&lt;','>':'&gt;','"':'&quot;',"'":'&#39;'}[c])); }
loadLeadsUI();
